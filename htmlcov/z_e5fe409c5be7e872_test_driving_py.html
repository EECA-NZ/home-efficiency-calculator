26def test_solar_friendly_ev_charging_profile_minimal_daily_need(): 

27 """ 

28 Test the scenario where daily needed charging hours are so small 

29 that it should all fit into the solar window on a solar-enabled day. 

30 For Tuesday (2019-01-01), we expect no spillover into night hours. 

31 """ 

32 annual_kwh = 73.0 # ~0.2 kWh/day 

33 charger_kw = 7.0 # => ~0.0286 hours/day (under 2 minutes) 

34 arr = solar_friendly_ev_charging_profile(annual_kwh, charger_kw, year=2019) 

35 

36 # Check shape & sum 

37 assert arr.shape[0] == 8760 

38 np.testing.assert_almost_equal(arr.sum(), 1.0, decimal=6) 

39 

40 # Convert array back to a time-series to examine day-of-week usage 

41 idx = pd.date_range("2019-01-01 00:00:00", "2019-12-31 23:00:00", freq="h") 

42 profile = pd.Series(arr, index=idx) 

43 

44 # Jan 1, 2019 was a Tuesday => solar window: 13:00–16:00 (3 hours) 

45 # We'll slice exactly that day 

46 day_slice = profile.loc["2019-01-01"] 

47 # Use between_time instead of partial-string slicing 

48 solar_period = day_slice.between_time("13:00", "15:59") 

49 solar_hours = solar_period.sum() 

50 

51 # Now check the night window for that same day: 

52 night_period = day_slice.between_time("21:00", "23:59") 

53 night_hours = night_period.sum() 

54 

55 assert solar_hours > 0, "Expected some usage in the solar window on Tuesday." 

56 assert night_hours == 0, "Should not need any night charging if solar is enough."
