1""" 

2Space heating profile calculation with heat pump COP logic. 

3 

4We compute an hourly space heating demand profile for the postcode, 

5factoring in day-of-week scheduling and optional heat pump COP. 

6 

7Demand is computed as: max(setpoint - T_outside, 0). 

8There is always heating in two baseline windows per day: 

9 - Morning: 7am–9am 

10 - Evening: 5pm–9pm 

11And optionally an "all-day" window (7am–9pm) for a certain number 

12of days per week, determined by the 'heating_during_day' parameter: 

13 

14 - "Never" => 0 full-day heating days per week 

15 - "1-2 days a week" => 1 day in odd weeks, 2 in even weeks 

16 - "3-4 days a week" => 3 days in odd weeks, 4 in even weeks 

17 - "5-7 days a week" => 5 days in odd weeks, 7 in even weeks 

18 

19If main_heating_source == "Heat pump", we reduce the required 

20energy by dividing by a COP timeseries. By default ("constant"), 

21this COP is taken from a dictionary keyed by climate zone. 

22If cop_calculation == "scaled_carnot_cop", we estimate a 

23scaled Carnot COP based on (21°C - T_outside). 

24 

25Finally, we normalize to produce a shape factor summing to 1. 

26"""

48def _days_for_week(week_num: int, option: str) -> int: 

49 """ 

50 Return how many "full-day heating" days are used in a given ISO week. 

51 

52 Parameters 

53 ---------- 

54 week_num : int 

55 The ISO week number (1..52 or 53). 

56 option : str 

57 One of: "Never", "1-2 days a week", "3-4 days a week", "5-7 days a week". 

58 

59 Returns 

60 ------- 

61 int 

62 Number of days in the given week for which the user wants 

63 all-day heating (7am–9pm). 

64 """ 

65 if option == "1-2 days a week": 

66 return 1 if (week_num % 2 == 1) else 2 

67 if option == "3-4 days a week": 

68 return 3 if (week_num % 2 == 1) else 4 

69 if option == "5-7 days a week": 

70 return 5 if (week_num % 2 == 1) else 7 

71 return 0 # Default for "Never"

74def _get_heating_cop_series( 

75 postcode: str, 

76 temperature_series: pd.Series, 

77 cop_calculation: str = "constant", 

78 setpoint: float = SPACE_HEATING_SETPOINT, 

79) -> pd.Series: 

80 """ 

81 Return an hourly COP series for space heating. 

82 

83 If cop_calculation == "constant", we use a single 

84 climate-zone-specific COP from HEAT_PUMP_COP_BY_CLIMATE_ZONE, 

85 repeated for every hour. 

86 

87 If cop_calculation == "scaled_carnot_cop", we use 

88 carnot_cop(T_hot=setpoint, T_cold=outside_temperature) per hour 

89 and scale it so that its average matches the climate zone's 

90 annual COP from the dictionary. 

91 

92 Parameters 

93 ---------- 

94 postcode : str 

95 The postcode to derive the climate zone from. 

96 temperature_series : pd.Series 

97 Hourly outside temperatures (°C). 

98 cop_calculation : str 

99 "constant" or "scaled_carnot_cop". 

100 setpoint : float 

101 Desired indoor temperature in °C for Carnot COP calculation 

102 (default=21.0). 

103 

104 Returns 

105 ------- 

106 pd.Series 

107 A timeseries of COP values (>=1). 

108 """ 

109 cz = climate_zone(postcode) 

110 annual_cop = HEAT_PUMP_COP_BY_CLIMATE_ZONE[cz] 

111 

112 if cop_calculation == "constant": 

113 # Single numeric COP repeated 

114 return pd.Series(annual_cop, index=temperature_series.index, name="COP") 

115 

116 if cop_calculation == "scaled_carnot_cop": 

117 # Compute the theoretical Carnot COP for each hour, then scale 

118 hourly_cop = temperature_series.apply(lambda t: carnot_cop(setpoint, t)) 

119 # Scale so that the mean matches annual_cop 

120 scale_factor = annual_cop / hourly_cop.mean() 

121 return (hourly_cop * scale_factor).rename("COP") 

122 

123 raise ValueError(f"Unknown cop_calculation: {cop_calculation}")

127def space_heating_profile( 

128 postcode: str, 

129 heating_during_day: str, 

130 setpoint: float = SPACE_HEATING_SETPOINT, 

131 main_heating_source: str = "Heat pump", 

132 cop_calculation: str = "constant", 

133) -> pd.Series: 

134 """ 

135 Compute an hourly space heating demand profile for the specified postcode. 

136 

137 Steps: 

138 1. Retrieve hourly outside temperatures via hourly_ta(postcode). 

139 2. Compute raw heating demand = max(setpoint - temperature, 0). 

140 3. If main_heating_source == "Heat pump", build a COP timeseries 

141 (either "constant" or "scaled_carnot_cop") and divide raw demand 

142 by that COP to get net electric demand. Otherwise, net electric 

143 demand = raw demand. 

144 4. Determine day scheduling: "baseline" (7–9am, 5–9pm) or 

145 "full-day" (7am–9pm) on certain days each week. 

146 5. Apply the active schedule to the net demand. 

147 6. Normalize so sum(demand) == 1. 

148 

149 The number of "full-day" heating days per week is determined by 

150 'heating_during_day', which can be: 

151 - "Never" 

152 - "1-2 days a week" 

153 - "3-4 days a week" 

154 - "5-7 days a week" 

155 

156 For each ISO week (week_num), we pick that many days 

157 in ascending date order to be "full-day" (7am–9pm). 

158 The rest remain on the baseline schedule. 

159 

160 Parameters 

161 ---------- 

162 postcode : str 

163 The postcode used by hourly_ta() to retrieve outside temps 

164 and by climate_zone() to retrieve the climate zone. 

165 heating_during_day : str 

166 One of: "Never", "1-2 days a week", "3-4 days a week", "5-7 days a week". 

167 setpoint : float, optional 

168 Desired indoor temperature in °C (default=21.0). 

169 main_heating_source : str, optional 

170 "Heat pump", "Resistive", etc. If "Heat pump", we reduce energy by the COP. 

171 cop_calculation : str, optional 

172 "constant" (use dictionary value per climate zone) 

173 or "scaled_carnot_cop" (compute scaled carnot for each hour). 

174 

175 Returns 

176 ------- 

177 pd.Series 

178 A Series of length 8760 (non-leap year) with an hourly DateTimeIndex, 

179 representing the normalized space heating demand (kWh fraction). 

180 Its sum is 1.0. 

181 """ 

182 # 1. Retrieve outside temperature data 

183 temperature_series = hourly_ta(postcode).copy() 

184 

185 # 2. Compute raw heating demand (thermal) for each hour 

186 demand_raw = np.maximum(setpoint - temperature_series, 0.0) 

187 

188 # 3. If it's a heat pump, build a COP timeseries and divide 

189 # otherwise, net demand = demand_raw 

190 if main_heating_source.lower() == "heat pump": 

191 cop_series = _get_heating_cop_series( 

192 postcode, temperature_series, cop_calculation, setpoint 

193 ) 

194 net_demand = demand_raw / cop_series 

195 else: 

196 net_demand = demand_raw 

197 

198 # 4. For scheduling, store net_demand in a DataFrame 

199 df = pd.DataFrame( 

200 { 

201 "net_demand": net_demand, 

202 "temperature": temperature_series, 

203 }, 

204 index=temperature_series.index, 

205 ) 

206 

207 df["date"] = df.index.normalize() 

208 iso_data = df.index.isocalendar() 

209 df["week"] = iso_data["week"] 

210 df["dayofweek"] = df.index.weekday # Monday=0, Sunday=6 

211 

212 # 5. Mark full-day vs baseline days 

213 full_heating_dates = {} 

214 grouped = df.groupby("week") 

215 for week_val, group in grouped: 

216 unique_dates = np.sort(group["date"].unique()) 

217 n_full = _days_for_week(week_val, heating_during_day) 

218 for i, day_val in enumerate(unique_dates): 

219 full_heating_dates[day_val] = i < n_full 

220 

221 df["full_heating"] = df["date"].map(full_heating_dates).fillna(False) 

222 

223 # 6. Build boolean masks for each schedule: 

224 hours = df.index.hour 

225 

226 # Full day window: (7, 21) 

227 mask_full_day = (hours >= FULL_DAY_WINDOW[0]) & (hours < FULL_DAY_WINDOW[1]) 

228 

229 # Baseline window is union of two intervals: (7,9) and (17,21) 

230 mask_baseline = np.zeros(len(df), dtype=bool) 

231 for start, end in BASELINE_WINDOWS: 

232 mask_baseline |= (hours >= start) & (hours < end) 

233 

234 # Apply schedule: pick mask_full_day if full_heating==True, else mask_baseline 

235 schedule_mask = np.where(df["full_heating"], mask_full_day, mask_baseline) 

236 net_scheduled = df["net_demand"] * schedule_mask.astype(float) 

237 

238 # 7. Normalize sum to 1 

239 total = net_scheduled.sum() 

240 if total > 0: 

241 net_scheduled /= total 

242 

243 logger.info("HERE IN HEATING USAGE PROFILE HELPERS") 

244 

245 # Return as a pandas Series with DateTimeIndex, normalized to sum to 1 

246 return pd.Series(net_scheduled, index=df.index, name="heating_profile")
