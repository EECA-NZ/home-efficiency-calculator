17def heating_frequency_factor(heating_days_per_week): 

18 """ 

19 Calculate the heating frequency factor based on the number of heating days per week. 

20 Assumes that heating occurs every morning and evening. 

21 Daytime heating is based on the response to the question 

22 "How often do you heat your home during the day?". 

23 

24 Parameters 

25 ---------- 

26 heating_days_per_week : float 

27 The number of days per week that heating occurs. 

28 

29 Returns 

30 ------- 

31 float 

32 The heating frequency factor. 

33 """ 

34 heating_mornings = 7 

35 heating_evenings = 7 

36 return ( 

37 HEATING_PERIOD_FACTOR["Morning (per day)"] * heating_mornings 

38 + HEATING_PERIOD_FACTOR["Day (per day)"] * heating_days_per_week 

39 + HEATING_PERIOD_FACTOR["Evening (per day)"] * heating_evenings 

40 )

43def add_gst(plan: BaseModel) -> BaseModel: 

44 """ 

45 Adjust all cost-related fields in a plan by adding 15% GST. 

46 Don't alter the original plan object but manipulate a copy. 

47 """ 

48 gst_rate = 1.15 

49 plancopy = plan.model_copy() 

50 for field, value in plan.model_dump().items(): 

51 if isinstance(value, dict): 

52 # Apply GST to each value in the dictionary 

53 adjusted_dict = {k: v * gst_rate for k, v in value.items()} 

54 setattr(plancopy, field, adjusted_dict) 

55 elif "charge" in field or "per_" in field: 

56 # Apply GST to flat rate fields 

57 setattr(plancopy, field, value * gst_rate) 

58 return plancopy

61def round_floats_to_2_dp(dictionary): 

62 """ 

63 Round all floats in a dictionary to 2 decimal places. 

64 Recursively rounds floats in nested dictionaries. 

65 """ 

66 for key, value in dictionary.items(): 

67 if isinstance(value, float): 

68 dictionary[key] = round(value, 2) 

69 elif isinstance(value, dict): 

70 round_floats_to_2_dp(value) 

71 return dictionary

74def safe_percentage_reduction(current: float, alternative: float) -> float: 

75 """ 

76 Safely calculate percentage reduction from current to alternative values. 

77 Handles cases where current is zero to avoid division by zero errors. 

78 """ 

79 if current == 0: 

80 return np.nan if alternative != 0 else 0 

81 return 100 * (current - alternative) / current
